set(SRCS
    tricore_allocateheap.c
    tricore_cache.c
    tricore_checkstack.c
    tricore_createstack.c
    tricore_csa.c
    tricore_exit.c
    tricore_getintstack.c
    tricore_idle.c
    tricore_initialize.c
    tricore_initialstate.c
    tricore_irq.c
    tricore_main.c
    tricore_mdelay.c
    tricore_nputs.c
    tricore_perf.c
    tricore_registerdump.c
    tricore_releasestack.c
    tricore_saveusercontext.c
    tricore_stackframe.c
    tricore_svcall.c
    tricore_systimer.c
    tricore_tcbinfo.c
    tricore_trapcall.c
    tricore_usestack.c
    tricore_fork.c)

if(NOT CONFIG_DISABLE_SIGNALS)
  list(APPEND SRCS tricore_schedulesigaction.c tricore_sigdeliver.c)
endif()

if(CONFIG_BUILD_PROTECTED OR CONFIG_BUILD_KERNEL)
  list(APPEND SRCS tricore_task_start.c tricore_pthread_start.c)

  if(NOT CONFIG_DISABLE_SIGNALS)
    list(APPEND SRCS tricore_signal_dispatch.c)
  endif()
  if(CONFIG_BUILD_PROTECTED)
    target_sources(arch_interface PRIVATE tricore_signal_handler.c)
  endif()
  if(CONFIG_ARCH_PERF_EVENTS_USER_ACCESS)
    target_sources(arch_interface PRIVATE tricore_perf.c)
  endif()
endif()

if(CONFIG_SCHED_BACKTRACE)
  list(APPEND SRCS tricore_backtrace.c)
endif()

if(CONFIG_ARCH_USE_MPU)
  list(APPEND SRCS tricore_mpu.c)
endif()

if(CONFIG_ARCH_ADDRENV)
  list(APPEND SRCS tricore_addrenv.c)
endif()

set(IFXFLAGS -DIFX_CFG_EXTEND_TRAP_HOOKS -DIFX_USE_SW_MANAGED_INT)

target_sources(arch PRIVATE ${SRCS})
nuttx_add_aux_library(STARTUP_OBJS crt0.c)
target_compile_options(
  STARTUP_OBJS
  PRIVATE
    $<GENEX_EVAL:$<TARGET_PROPERTY:nuttx_global,NUTTX_ELF_APP_COMPILE_OPTIONS>>)
target_sources(nuttx PRIVATE tricore_doirq.c)
target_compile_options(arch PRIVATE ${IFXFLAGS})
target_compile_options(nuttx PRIVATE ${IFXFLAGS})
target_include_directories(
  nuttx SYSTEM PRIVATE ${CMAKE_SOURCE_DIR}/include ${CMAKE_BINARY_DIR}/include
                       ${CMAKE_BINARY_DIR}/include_arch)

if(CONFIG_ARCH_TOOLCHAIN_TASKING)
  target_sources(nuttx PRIVATE tricore_cxx_initialize.c)
  target_link_options(nuttx PRIVATE -OtxYcL)
  if(CONFIG_BUILD_PROTECTED)
    target_sources(nuttx_user PRIVATE tricore_cxx_initialize.c)
    target_link_options(nuttx_user PRIVATE --optimize=0)
  endif()
endif()
